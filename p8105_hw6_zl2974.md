set.seed(1)
weather_bt =
  modelr::bootstrap(weather_df,
                    n = 1000,
                    id = "id") %>%
  mutate(
    strap = map(strap, as_tibble),
    md =
      map(.x = strap,
          ~ lm(tmax ~ tmin, data = .x)),
    r_square =
      map2_dbl(.x = md,

               ~ modelr::rsquare(.x, .y)),
